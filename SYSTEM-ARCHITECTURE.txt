┌────────────────────────────────────────────────────────────────────────────┐
│ 🚀 FLUX D'UTILISATION                                                      │
└────────────────────────────────────────────────────────────────────────────┘

  1. INSCRIPTION
     ┌─────────────────────────────────────────┐
     │ auth-login.php                          │
     │ Remplir formulaire → POST auth.php      │
     │ ↓                                       │
     │ auth.php vérifie & crée utilisateur     │
     │ ↓                                       │
     │ 10 progress rows creées (stages 1-10)   │
     │ ↓                                       │
     │ 1 rewards row creée                     │
     │ ↓                                       │
     │ SESSION['user_id'] créée                │
     │ ↓                                       │
     │ REDIRECTION → dashboard-stages.php ✅   │
     └─────────────────────────────────────────┘

  2. CONNEXION
     ┌─────────────────────────────────────────┐
     │ auth-login.php (form login)             │
     │ Envoyer email + password → auth.php     │
     │ ↓                                       │
     │ auth.php vérifie password avec bcrypt   │
     │ ↓                                       │
     │ SESSION créée                           │
     │ ↓                                       │
     │ REDIRECTION → dashboard ✅              │
     └─────────────────────────────────────────┘

  3. JOUER UNE ÉTAPE
     ┌─────────────────────────────────────────┐
     │ Dashboard → Click Stage → stage.php     │
     │ ?stage=1                                │
     │ ↓                                       │
     │ stage.php charge stage-1-tunisia.html   │
     │ + injecte api-manager.js                │
     │ ↓                                       │
     │ Utilisateur joue                        │
     │ ↓                                       │
     │ api-manager.saveProgress()              │
     │ → progress_api.php?action=save_progress │
     │ ↓                                       │
     │ Données sauvegardées en DB ✅           │
     └─────────────────────────────────────────┘

  4. FERMER & ROUVRIR
     ┌─────────────────────────────────────────┐
     │ Utilisateur ferme navigateur            │
     │ ↓                                       │
     │ Données restent en DB ✅                │
     │ ↓                                       │
     │ Utilisateur rouvre navigateur           │
     │ ↓                                       │
     │ Se connecte à auth-login.php            │
     │ ↓                                       │
     │ Va à dashboard                          │
     │ ↓                                       │
     │ loadProgress() depuis API               │
     │ ↓                                       │
     │ MÊME ÉTAPE ET PROGRESSION ! ✅          │
     └─────────────────────────────────────────┘


┌────────────────────────────────────────────────────────────────────────────┐
│ 💾 SCHÉMA DE LA BASE DE DONNÉES                                            │
└────────────────────────────────────────────────────────────────────────────┘

  DATABASE: mondo_magique
  ═════════════════════════════════════════════════════════════════════════

  1. TABLE: users
     ┌─────────────────────────────────────┐
     │ id INT PRIMARY KEY AUTO_INCREMENT   │
     │ username VARCHAR UNIQUE             │
     │ email VARCHAR UNIQUE                │
     │ password VARCHAR (bcrypt hashed)    │
     │ created_at TIMESTAMP                │
     │ last_login TIMESTAMP                │
     └─────────────────────────────────────┘

  2. TABLE: progress
     ┌─────────────────────────────────────┐
     │ user_id → users.id FK               │
     │ stage_num INT (1-10)                │
     │ completed BOOLEAN                   │
     │ qcm_score INT (0-100)               │
     │ essay_score INT (0-100)             │
     │ diamonds INT                        │
     │ coins INT                           │
     │ current_step INT (1-6)              │
     │ updated_at TIMESTAMP                │
     └─────────────────────────────────────┘

  3. TABLE: rewards
     ┌─────────────────────────────────────┐
     │ user_id → users.id FK UNIQUE        │
     │ total_diamonds INT                  │
     │ total_coins INT                     │
     │ total_stages_completed INT          │
     └─────────────────────────────────────┘

  4. TABLE: qcm_answers
     ┌─────────────────────────────────────┐
     │ user_id → users.id FK               │
     │ stage_num INT (1-10)                │
     │ q1, q2, q3, q4, q5 VARCHAR         │
     │ q1_correct, q2_correct, etc. BOOL   │
     │ PRIMARY KEY (user_id, stage_num)    │
     └─────────────────────────────────────┘

  5. TABLE: essays
     ┌─────────────────────────────────────┐
     │ user_id → users.id FK               │
     │ stage_num INT (1-10)                │
     │ content LONGTEXT                    │
     │ word_count INT                      │
     │ score INT (0-100)                   │
     │ PRIMARY KEY (user_id, stage_num)    │
     └─────────────────────────────────────┘


┌────────────────────────────────────────────────────────────────────────────┐
│ 🔗 API REST ENDPOINTS                                                      │
└────────────────────────────────────────────────────────────────────────────┘

  AUTHENTIFICATION
  ═════════════════════════════════════════════════════════════════════════
  POST   /auth.php?action=register    → Créer compte
  POST   /auth.php?action=login       → Connexion
  GET    /auth.php?action=logout      → Déconnexion
  GET    /auth.php?action=check_session → Vérifier session
  GET    /auth.php?action=get_user_data → Récupérer données

  PROGRESSION
  ═════════════════════════════════════════════════════════════════════════
  POST   /progress_api.php?action=save_progress → Sauver étape
  GET    /progress_api.php?action=load_progress → Charger étape
  GET    /progress_api.php?action=load_all_progress → Charger tout
  POST   /progress_api.php?action=save_qcm     → Sauver QCM
  POST   /progress_api.php?action=save_essay   → Sauver essai
  POST   /progress_api.php?action=complete_stage → Terminer étape
  GET    /progress_api.php?action=get_rewards  → Récompenses
